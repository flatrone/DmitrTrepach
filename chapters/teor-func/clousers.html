Счетчик на замыканиях

function test() {
	let num = 1;
	
	return function() {
		alert(num);
	}
}

let func = test();


Для более понятного восприятия, вынесем переменную num в голобал

let num = 1;

function test() {
	return function() {
		alert(num);
		num++;
	};
};

let func1 = test();


Для еще более легкого понятия, уберем возврощаемую функцию

let num = 1;

function test() {
        alert(num);
        num++;
};

let func1 = test;
