Т.к. функция возвращает функцию, можно ее вазвать сразу на месте, вот-так func()()
    function func() {
        return function() {
            return '!';
        };
    }
    
    alert( func()() ); // выведет '!'


    любой уровень ложенности

    function func() {
        return function() {
            return function() {
                return '!';
            };
        };
    }
    
    alert( func()()() );


    Параметры

    function func(num1) {
        return function(num2) {
            return num1 + num2;
        };
    }
    
    alert( func(1)(2) )
