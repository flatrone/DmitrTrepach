1 решение 

    function parent() {
        let self = this; // запишем this в любую переменную, например, в self
        function child() {
            отсюда доступен self 
        }
    }



    2 решение
    Передача параметром

    function parent() {
        child(this); // передаем параметром this
        
        function child(param) {
            // в переменной param лежит переданное содержимое this
        }
    }



    3 решение
    Не имеют своего контекста, но сохранают контекст родителя

    
function parent() {
	console.log(this.value); // выведет 'text'
	
	let child = () => {
		console.log(this.value); // выведет 'text'
	}
	child();
}
