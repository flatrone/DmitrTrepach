Флаги

Для проверки отсутвия чего-либо


Плохо

let arr = [1, 2, 3, 4, 3, 5];

for (let elem of arr) {
    if (elem == 3) {
        console.log('есть'); // выведет несколько раз
    }
}
    

рабочий метод

let arr = [1, 2, 3, 4, 3, 5];

for (let elem of arr) {
    if (elem == 3) {
        console.log('есть');
        break;  // завершим цикл
    }
}
    
        
Лучший метод

let arr = [1, 2, 3, 4, 5];
let flag = false;

for (let elem of arr) {
	if (elem == 3) {
		flag = true;
		break;
	}
}
